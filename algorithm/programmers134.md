##  문자열 계산하기

```JavaScript
const solution = (my_string) => my_string.split(' ').reduce((obj, cur, index, origin) => {
    let {acc, isPlus} = obj;
    if (cur === '+') {
        isPlus = true;
    } else if (cur === '-') {
        isPlus = false;
    } else if (isPlus === true) {
        acc += Number(cur);
    } else if (isPlus === false) {
        acc -= Number(cur);
    } else {
        acc += Number(cur);
    }
    return {acc, isPlus};
}, {acc: 0, isPlus: null}).acc;
```
