# 3. fetch

**오늘 사용한 fetch의 코드 설명**

```React
// 회원 가입
const signupAPI = (id, pwd, pwd_check) => {
  return function (dispatch, getState, { history }) {
    const API = "서버 아이피/user/signup";
// 포스트 방식으로 보내기
    fetch(API, {
      method: "POST",
      headers: {
        // 컨텐트 타입은 서버와 맞추기
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        // 보내는 JSON 양식 맞추기
        id: id,
        password: pwd,
        passwordConfirm: pwd_check,
      })
    })
      .then((response) => response)
      .then((result) => {
        dispatch(setUser({
          id: id,
        }))
        console.log(result);
        window.alert('회원가입 되었습니다.')
        history.push('/')
      })
  }
}
```
